'number' + 3 + 3	is result: number33
	//происходит конкатенация строк
	

null + 3	is result: 3
	//происходит конвертация null в число 0
	0+3=3

5 && "qwerty" 	     is result: qwerty
	// оператор && выводит первым значение false или последнее значение если все true
	   поскольку 5 это true, тогда выведется последнее значение, которое соответствует "qwerty"


+'40' + +'2' + "hillel"		is result: 42hillel
	//оператор + конвертирует значение '40' в number и '2' в number
	  происходит сложение чисел 40+2
	  число 42  конвертируется в string
	  происходит конкатенация строк '42'+ "hillel"
	  			

'10' - 5 === 6		is result: false
	//'10' конвертируется в number
	  10-5=5
	  5 строго не равно 6 это false

true + false		is result: 1
	// JS конвертирует значения boolean в числа
	  поскольку true=1, а false=0 происходит сложение чисел
	  1+0=1

'4px' - 3	      is result: NaN
	// string '4px' не может быть конвертирована в число, значит результат NaN
	  NaN -3 дает результат NaN

'4' - 3		is result: 1
	// string '4' конвертируется в число
	  4-3=1

'6' + 3 ** 0		is result: 61
	// оператор ** обозначает степень
	  3 ** 0 = 3 в нулевой степени = 1
	  при знаке + происходит конкатенация строк

12 / '6'	is result: 2
	// string '6' конвертируется в число
	  12/6=2

'10' + (5 === 6)	is result: 10false
	// первое действие в (5===6) является false
	  значение false конвертируется в string
	  происходит конкатенация '10'+ 'false'

null == '' 	is result: false
	// это выражение оценивается как false, потому что они не приводятся к одному значению

3 ** (9 / 3)	is result: 27
	// первое действие в скобках 9/3=3
	  3**3 = 3 в третьей степени = 27

!!'false' == !!'true'	      is result: true
	// выражение !!'false' конвертируется в 'false'
	   выражение !!'true' конвертируется в 'true'
	   соответственно 'false'=='true'

0 || '0' && 1 		is result: 1
	// первым идет оператор сравнения И (&&)
	   '0' && 1 в этом выражение правильным будет последнее значение true
	   следующее выражение 0||1 в котором 0 будет false, а 1-true

(+null == false) < 1	      is result: false
	// +null конвертируется в число 0
	   0==false это 0=0 соответствует true
	   true	< 1 это как 1< 1 и это false

false && true || true 	        is result: true
	// false && true будет false, поскольку этот оператор возвращает первый false результат
	  false||true оператор возвращает первый true результат

false && (false || true)           is result: false
	// false || true оператор вернет первый true результат
	   false && true оператор вернет первый false результат

(+null == false) < 1 ** 5	     is result: false
	// +null конвертирует в число 0
	   0==false или  0==0 - это true
	   1 в 5 степени=1
	   true	< 1 или 1< 1 это false
